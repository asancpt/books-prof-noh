MODEL
 COMM
  NDER 1
  NPAR 4 
  NSEC 5
  NCON 2
  PNAM 'VF','KA', 'SYNT', 'CL'
  SNAM 'Css', 'T12', 'Tss', 'MRT', 'MRT2'
END

TEMP
DOSE1=CON(1)
TLAG=CON(2)
T=X
IF T LE TLAG THEN
FINF =0
ELSE
FINF=DOSE1*KA*DEXP(-KA*(T-TLAG))
ENDIF
END
START
Z(1)=SYNT/CL
END
DIFF
DZ(1)=(FINF+SYNT-CL*Z(1))/VF
END
FUNC 1
F=Z(1)
END
SECO
CSS=SYNT/CL
T12=0.693*VF/CL
TSS=3.5*T12
MRT1=VF/CL
MRT2=1/(CL/VF) - 1/KA
END
EOM
